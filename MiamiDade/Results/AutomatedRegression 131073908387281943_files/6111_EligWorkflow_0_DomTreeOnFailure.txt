	function ChangeDashboardChart(chartsPanelId, isNext) {
		var charts = $(chartsPanelId).find(".js-dashboard-chart");
		var newChart = charts[0];
		for (var i = 0; i < charts.length; i++) {
			var currentChart = $(charts[i]);
			if (currentChart.css('display') != 'none') {
				newChart =
					isNext ?
						(i + 1 < charts.length) ? charts[i + 1] : charts[0] :
						(i > 0) ? charts[i - 1] : charts[charts.length - 1];
				currentChart.hide();
			}
		}
		$(newChart).show();
		return false;
	}
